29class UNetModel(nn.Module):
#

  • channel_multipliers 은 각 레벨의 채널 수에 대한 곱셈 요소입니다.

  • tf_layers 은 트랜스포머의 트랜스포머 층 수

입력이 다른 모듈을 위한 순차 블록

이 순차 모듈은 ResBlock , nn.Conv 및 SpatialTransformer 와 같은 다양한 모듈로 구성될 수 있으며 일치하는 서명으로 호출 할 수 있습니다.

185class TimestepEmbedSequential(nn.Sequential):
# 
193    def forward(self, x, t_emb, cond=None):
194        for layer in self:
195            if isinstance(layer, ResBlock):
196                x = layer(x, t_emb)
197            elif isinstance(layer, SpatialTransformer):
198                x = layer(x, cond)
199            else:
200                x = layer(x)
201        return x
