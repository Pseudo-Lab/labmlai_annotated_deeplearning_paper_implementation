70    def init(self):
#

කාර්වානා දත්ත කට්ටලය ආරම්භ කරන්න 

72        self.dataset = CarvanaDataset(lab.get_data_path() / 'carvana' / 'train',
73                                      lab.get_data_path() / 'carvana' / 'train_masks')
#

ආකෘතියආරම්භ කරන්න

79                                                       shuffle=True, pin_memory=True)
#

ප්රශස්තකරණයසාදන්න

අහඹුනියැදියක් ලබා ගන්න 

93        x, _ = self.dataset[np.random.randint(len(self.dataset))]

98        mask = self.sigmoid(self.model(x[None, :]))
#

වෙස්මුහුණෙහි ප්රමාණයට රූපය වගා කරන්න

104    def train(self):
#

දත්තකට්ටලය හරහා නැවත කරන්න. එක් mix 50 යුගයකට නියැදි වේලාවන් භාවිතා කරන්න.

පුරෝකථනයකරන ලද වෙස්මුහුණු පිවිසුම් ලබා ගන්න 

121            logits = self.model(image)
#

ඉලක්කගතවෙස්මුහුණ පිවිසුම් ප්රමාණයට වගා කරන්න. යූ-නෙට් හි සංවලිත ස්ථර වල අපි පෑඩින් භාවිතා නොකරන්නේ නම් පිවිසුම් ප්රමාණය කුඩා වේ. 

124            mask = torchvision.transforms.functional.center_crop(mask, [logits.shape[2], logits.shape[3]])

ප්රශස්තිකරණපියවරක් ගන්න 

130            self.optimizer.step()

අලාභයලුහුබඳින්න
